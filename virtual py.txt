# Activate the virtual environment  phyton
source ~/myenv/bin/activate
cd /var/www/html/ai/
deactivate

#جدید (newChatbotPanel)
cd /var/www/html/ai/backend
source venv/bin/activate
uvicorn app.main:app --reload --host 0.0.0.0 --port 5000

ثبت کاربر جدید
curl -X POST http://localhost:5000/api/register -H "Content-Type: application/json" -d '{"email": "admin@example.com", "password": "admin123"}'

ورود و دریافت توکن
curl -X POST http://localhost:5000/api/token -H "Content-Type: application/x-www-form-urlencoded" -d "username=admin@example.com&password=admin123"

درخواست کراول کردن سایت
curl -X POST "http://localhost:5000/api/websites/crawl" -H "Authorization: Bearer {TOKEN}" -H "Content-Type: application/json" -d '{"url": "https://parsicanada.com", "max_pages": 50}'

بررسی سایت کراول شده
curl -X GET http://localhost:5000/api/websites/{ID} -H "Authorization: Bearer {TOKEN}"
سوال از LLM
curl -X POST "http://localhost:5000/api/chat" -H "Authorization: Bearer {TOKEN}" -H "Content-Type: application/json" -d '{"website_id": 34, "message": "لطفاً در مورد خدمات مهاجرت به کانادا توضیح دهید", "use_rag": true, "use_llm": true}'


#فرانت

cd /var/www/html/ai/frontend
npm install
npm start
